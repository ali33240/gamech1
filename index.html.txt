        function playSound(type) {
            initAudio();
            const osc = audioCtx.createOscillator();
            const gainNode = audioCtx.createGain();
            osc.connect(gainNode);
            gainNode.connect(audioCtx.destination);
            if (type === 'correct') {
                osc.type = 'sine'; osc.frequency.setValueAtTime(659.25, audioCtx.currentTime); osc.frequency.setValueAtTime(783.99, audioCtx.currentTime + 0.1);
                gainNode.gain.setValueAtTime(0.1, audioCtx.currentTime); gainNode.gain.exponentialRampToValueAtTime(0.001, audioCtx.currentTime + 0.6);
                osc.start(); osc.stop(audioCtx.currentTime + 0.6);
            } else if (type === 'wrong') {
                osc.type = 'sawtooth'; osc.frequency.setValueAtTime(150, audioCtx.currentTime); osc.frequency.linearRampToValueAtTime(50, audioCtx.currentTime + 0.4);
                gainNode.gain.setValueAtTime(0.1, audioCtx.currentTime); gainNode.gain.linearRampToValueAtTime(0.001, audioCtx.currentTime + 0.4);
                osc.start(); osc.stop(audioCtx.currentTime + 0.4);
            }
        }

        function checkAnswer() {
            initAudio();
            if (!selectedInitial && !selectedFinal) {
                const box = document.querySelector('.bg-slate-50');
                box.classList.add('anim-shake'); setTimeout(()=>box.classList.remove('anim-shake'), 500); return;
            }
            const isInitialCorrect = (selectedInitial === targetInitial) || (!selectedInitial && !targetInitial);
            const isFinalCorrect = (selectedFinal === targetFinal);
            const isToneCorrect = (selectedTone.id === targetTone.id);

            if (isInitialCorrect && isFinalCorrect && isToneCorrect) {
                score += 10; document.getElementById('scoreVal').innerText = score;
                playSound('correct');
                showFeedback(true, targetChar);
                setTimeout(nextQuestion, 2500);
            } else {
                playSound('wrong');
                showFeedback(false);
                setTimeout(() => { closeFeedback(); playTargetSound(); }, 2000);
            }
        }
